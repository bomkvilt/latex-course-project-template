%%% Include a figure into a section
%   #1  - /fig - relative figure path
%   #2  - use a global label
%   #3  - the figure's lable (unique for all FIGURES in THE section)
%   #4  - place figure RIGTH HERE
%   #5  - the image's caption
%   #6  - settings forwards into a \includegraphics function
\NewDocumentCommand{\addFigure}{ m s o s o O{ }} {
    \bool_if:nTF {#4}
    { \begin{figure}[H]  }
    { \begin{figure}[ht] }
        \centering\includegraphics[#6]{#1}
        \IfValueT{#5} { \caption{#5} }
        \IfValueT{#3} { 
            \bool_if:nTF {#2}
            { \label{#3} }
            { \label{\sections@figPrefix#3} }
        }
    \end{figure}
}

% Create a reference on the section's figure
%   #1  - use a global label
%   #2  - a label was assigned to a figure during an insertion
\NewDocumentCommand{\rfg}{ s m } 
{   
    \bool_if:nTF {#1}
    { \secs_addReference:Nx{}{#2} }
    { \secs_addReference:Nx{\sections@figPrefix}{#2} }
}
