%   #1  - prefix to be added to all passed lables
%   #2  - list of labels

    \seq_clear:N \l@labels@full_seq    
    \clist_map_inline:nn { #2 }

        \seq_push:Nn \l@labels@full_seq {#1##1}
    }
    \cref{\seq_use:Nnnn \l@labels@full_seq {,}{,}{,}}
}
